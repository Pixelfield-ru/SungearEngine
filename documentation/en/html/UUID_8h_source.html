39  static std::string generateNew() noexcept
40  {
41  static std::random_device rd;
42  static std::mt19937_64 gen(rd());
43  static std::uniform_int_distribution<> dis(0, 15);
44  /*
45  * 1 0 0 0 = 8
46  * 1 0 0 1 = 9
47  * 1 0 1 0 = A
48  * 1 0 1 1 = B
49  */
50  static std::uniform_int_distribution<> dis2(8, 11);
51 
52  std::stringstream ss;
53  int i;
54  ss << std::hex;
55  for(i = 0; i < 8; i++)
56  {
57  ss << dis(gen);
58  }
59  ss << "-";
60  for(i = 0; i < 4; i++)
61  {
62  ss << dis(gen);
63  }
64  // 4 IS RANDOMLY OR PSEUDO-RANDOMLY GENERATED UUID
65  ss << "-4";
66  for(i = 0; i < 3; i++)
67  {
68  ss << dis(gen);
69  }
70  ss << "-";
71  ss << dis2(gen);
72  for(i = 0; i < 3; i++)
73  {
74  ss << dis(gen);
75  }
76  ss << "-";
77  for(i = 0; i < 12; i++)
78  {
79  ss << dis(gen);
80  };
81 
82  std::string uuid = ss.str();
83 
84  // std::cout << "uuid: " << m_uuid << std::endl;
85 
86  return uuid;
87  }
88  };
