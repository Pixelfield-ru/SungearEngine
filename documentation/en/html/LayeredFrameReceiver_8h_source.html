81  template<typename EffectT>
82  requires(std::is_base_of_v<PostProcessEffect, EffectT>)
83  void removeEffect() noexcept
84  {
85  std::erase_if(m_effects, [this](const Ref<PostProcessEffect>& otherEffect) {
86  if(SG_INSTANCEOF(otherEffect.get(), EffectT))
87  {
88  auto thisShared = shared_from_this();
89 
90  std::erase_if(otherEffect->m_parentPostProcessLayers, [this](const Weak<PostProcessLayer>& parentLayer) {
91  auto lockedLayer = parentLayer.lock();
92  return lockedLayer && lockedLayer.get() == this;
93  });
94 
95  otherEffect->onDetachFromLayer(thisShared);
96 
97  return true;
98  }
99 
100  return false;
101  });
102  }
