27  void includeFile(const std::shared_ptr<ShaderAnalyzedFile>& analyzedFile) noexcept
28  {
29  m_globalCode += analyzedFile->m_globalCode + "\n";
30 
31  for(const auto& subPassPair : analyzedFile->m_subPasses)
32  {
33  const auto& otherSubPassName = subPassPair.first;
34  const auto& otherSubPass = subPassPair.second;
35 
36  auto& thisSubPass = m_subPasses[otherSubPassName];
37 
38  thisSubPass.m_globalCode += otherSubPass.m_globalCode + "\n";
39 
40  for(const auto& otherSubShaderPair : otherSubPass.m_subShaders)
41  {
42  auto otherSubShaderType = otherSubShaderPair.first;
43  const auto& otherSubShader = otherSubShaderPair.second;
44 
45  Ref<SGSLESubShader> thisSubShader = addOrGetSubShader(otherSubPassName, otherSubShaderType);
46 
47  thisSubShader->m_code = otherSubShader->m_code + "\n" + thisSubShader->m_code;
48  }
49  }
50 
51  // m_linesTotalCount += analyzedFile.m_linesTotalCount;
52  m_includedFiles.push_back(analyzedFile);
53  }
54 
55  Ref<SGSLESubShader> addOrGetSubShader(const std::string& subPassName, SGSLESubShaderType subShaderType)
56  {
57  auto& subPass = m_subPasses[subPassName];
58  if(subPass.m_subShaders.find(subShaderType) == subPass.m_subShaders.end())
59  {
60  Ref<SGSLESubShader> newSubShader = MakeRef<SGSLESubShader>();
61  subPass.m_subShaders[subShaderType] = newSubShader;
62  newSubShader->m_code = m_globalCode + "\n" + subPass.m_globalCode;
63 
64  return newSubShader;
65  }
66  else
67  {
68  return subPass.m_subShaders[subShaderType];
69  }
70  }
