47  template<typename ScalarT = float>
48  static void rayAABBIntersection(const glm::vec<3, ScalarT, glm::defaultp>& rayOrigin,
49  const glm::vec<3, ScalarT, glm::defaultp>& rayDirection,
50  const glm::vec<3, ScalarT, glm::defaultp>& aabbMin,
51  const glm::vec<3, ScalarT, glm::defaultp>& aabbMax,
52  RayIntersectionInfo<ScalarT>& intersectionInfo) noexcept
53  {
54  glm::vec<3, ScalarT, glm::defaultp> dirfrac;
55  dirfrac = 1.0f / rayDirection;
56 
57  // r.org is origin of ray
58  float t1 = (aabbMin.x - rayOrigin.x) * dirfrac.x;
59  float t2 = (aabbMax.x - rayOrigin.x) * dirfrac.x;
60 
61  float t3 = (aabbMin.y - rayOrigin.y) * dirfrac.y;
62  float t4 = (aabbMax.y - rayOrigin.y) * dirfrac.y;
63 
64  float t5 = (aabbMin.z - rayOrigin.z) * dirfrac.z;
65  float t6 = (aabbMax.z - rayOrigin.z) * dirfrac.z;
66 
67  float tmin = std::max(std::max(std::min(t1, t2), std::min(t3, t4)), std::min(t5, t6));
68  float tmax = std::min(std::min(std::max(t1, t2), std::max(t3, t4)), std::max(t5, t6));
69 
70  // if tmax < 0, ray (line) is intersecting AABB, but the whole AABB is behind us
71  // if tmin > tmax, ray doesn't intersect AABB
72  if (tmax < 0 || tmin > tmax)
73  {
74  intersectionInfo.m_hitDistance = tmax;
75  intersectionInfo.m_isIntersected = false;
76  return;
77  }
78 
79  intersectionInfo.m_hitDistance = tmin;
80  intersectionInfo.m_isIntersected = true;
81  }

112  template<typename ScalarT = float>
113  static void rayPlaneIntersection(const glm::vec<3, ScalarT, glm::defaultp>& rayOrigin,
114  const glm::vec<3, ScalarT, glm::defaultp>& rayDirection,
115  const glm::vec<3, ScalarT, glm::defaultp>& planeOrigin,
116  const glm::vec<3, ScalarT, glm::defaultp>& planeNormal,
117  RayIntersectionInfo<ScalarT>& intersectionInfo,
118  const ScalarT& epsilon = 1e-8)
119  {
120  ScalarT denom = glm::dot(planeNormal, rayDirection);
121 
122  if(std::abs(denom) > epsilon)
123  {
124  ScalarT dist = glm::dot(planeOrigin - rayOrigin, planeNormal) / denom;
125  if(dist > 0)
126  {
127  intersectionInfo.m_hitDistance = dist;
128  intersectionInfo.m_isIntersected = true;
129  return;
130  }
131  }
132 
133  intersectionInfo.m_hitDistance = 0;
134  intersectionInfo.m_isIntersected = false;
135  }
