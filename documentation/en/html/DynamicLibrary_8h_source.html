130  #ifdef PLATFORM_OS_WINDOWS
131  static std::string getLastWindowsError()
132  {
133  DWORD errorMessageID = GetLastError();
134 
135  LPSTR messageBuffer = nullptr;
136  // Ask Win32 to give us the string version of that message ID.
137  // The parameters we pass in, tell Win32 to create the buffer that holds the message for us (because we don't yet know how long the message string will be).
138  size_t size = FormatMessageA(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
139  NULL, errorMessageID, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), (LPSTR)&messageBuffer, 0, NULL);
140  // Copy the error message into a std::string.
141  std::string message = std::string(messageBuffer, size) + " (Code '" + std::to_string(errorMessageID) + "').";
142  // Free the Win32's string's buffer.
143  LocalFree(messageBuffer);
144 
145  return message;
146  }

151  // loadFunction helper
152  template<typename Result, typename... Args>
153  struct DLFuncLoader<Result(Args...)>
154  {
155  std::function<Result(Args...)> operator()(native_handler_t nativeHandler, const char* funcName, std::string& err) const noexcept
156  {
157  #ifdef PLATFORM_OS_LINUX
158  using func_t = Result(*)(Args...);
159  #elif defined(PLATFORM_OS_WINDOWS)
160  using func_t = Result(__cdecl *)(Args...);
161  #endif
162 
163  func_t func = nullptr;
164 
165  #ifdef PLATFORM_OS_LINUX
166  func = (func_t) dlsym(nativeHandler, funcName);
167  if(!func)
168  {
169  err = dlerror();
170  }
171  #elif defined(PLATFORM_OS_WINDOWS)
172  func = (func_t) GetProcAddress(nativeHandler, funcName);
173  if(!func)
174  {
175  err = DynamicLibrary::getLastWindowsError();
176  }
177  #endif
178 
179  return func;
180  }
181  };
